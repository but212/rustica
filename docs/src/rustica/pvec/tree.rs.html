1//! Persistent Vector Tree Module
2//!
3//! This module implements the core tree structure for the persistent vector.
4//!
5//! This module defines the Tree type, which is the main data structure
6//! for the persistent vector. It manages a root node and provides vector operations.
7//!
8//! The Tree type implements a Relaxed Radix Balanced (RRB) tree, which is a
9//! persistent data structure optimized for both random access and structural
10//! updates at the ends. It provides efficient index-based access, append, and
11//! prepend operations.
12
13use std::fmt::{self, Debug};
14use std::iter::FromIterator;
15
16use crate::pvec::memory::IndexCache;
17use crate::pvec::memory::{AllocationStrategy, BoxedCachePolicy, Chunk, ManagedRef, MemoryManager};
18use crate::pvec::node::{Node, NODE_BITS};

20/// A persistent vector implemented as a Relaxed Radix Balanced (RRB) tree.
21///
22/// The tree provides efficient indexing, updating, and appending operations
23/// with persistence guarantees. All operations create a new version of the
24/// tree without modifying the original, allowing for efficient structural
25/// sharing between different versions.
26#[derive(Clone)]
27pub(crate) struct Tree<T> {
28    /// The root node of the tree.
29    ///
30    /// This is wrapped in a managed reference to enable efficient sharing
31    /// between different versions of the tree.
32    root: ManagedRef<Node<T>>,
33
34    /// The number of elements in the tree.
35    ///
36    /// This is cached at the tree level to avoid traversing the entire
37    /// tree structure when size information is needed.
38    size: usize,
39
40    /// The height of the tree (number of levels).
41    ///
42    /// A height of 0 means the tree consists only of a single leaf node.
43    /// Each additional level adds a shift of NODE_BITS bits to the tree.
44    height: usize,
45
46    /// The memory manager used for allocating and recycling nodes.
47    ///
48    /// This handles memory management for nodes, chunks, and other
49    /// data structures used by the tree.
50    manager: MemoryManager<T>,
51
52    /// Cache for accelerating repeated access operations.
53    ///
54    /// The cache stores information about the most recently accessed index,
55    /// which can significantly speed up accesses to nearby indices.
56    cache: IndexCache,
57
58    /// Policy for controlling when to use the cache.
59    ///
60    /// This policy determines when the cache should be used for operations.
61    /// It can be set to control caching behavior dynamically.
62    cache_policy: BoxedCachePolicy,
63
64    /// The chunk size used for allocating chunks.
65    chunk_size: usize,
66}

68impl<T: Clone> Tree<T> {
69    /// Create a new, empty tree with the given chunk size.
70    #[inline(always)]
71    #[must_use]
72    pub fn new_with_chunk_size(chunk_size: usize) -> Self {
73        let manager = MemoryManager::new(AllocationStrategy::Direct);
74        let chunk = manager.allocate_chunk(Chunk::new_with_size(chunk_size));
75        let root = manager.allocate_node(Node::leaf(chunk));
76        Self {
77            root,
78            size: 0,
79            height: 0,
80            manager,
81            cache: IndexCache::new(),
82            cache_policy: Box::new(crate::pvec::memory::AlwaysCache),
83            chunk_size,
84        }
85    }
86
87    /// Create a new, empty tree with the default chunk size.
88    #[inline(always)]
89    #[must_use]
90    pub fn new() -> Self {
91        Self::new_with_chunk_size(crate::pvec::memory::DEFAULT_CHUNK_SIZE)
92    }

94    /// Create a new tree from a slice of elements, with the given chunk size.
95    #[inline(always)]
96    #[must_use]
97    pub fn from_slice_with_chunk_size(slice: &[T], chunk_size: usize) -> Self {
98        use crate::pvec::memory::Chunk;
99        use crate::pvec::node::{Node, NODE_SIZE};
100
101        if slice.is_empty() {
102            return Self::new_with_chunk_size(chunk_size);
103        }
104
105        // Step 1: Create leaf nodes from chunks
106        let mut leaf_nodes = Vec::new();
107        let mut size = 0;
108        let manager = MemoryManager::new(AllocationStrategy::Direct);
109        for chunk in slice.chunks(chunk_size) {
110            let mut c = Chunk::new_with_size(chunk_size);
111            for item in chunk {
112                c.push_back(item.clone());
113            }
114            size += c.len();
115            let chunk_ref = manager.allocate_chunk(c);
116            leaf_nodes.push(manager.allocate_node(Node::leaf(chunk_ref)));
117        }
118
119        // Step 2: Build tree from leaves upward
120        let mut current_level = leaf_nodes;
121        let mut height = 0;
122        while current_level.len() > 1 {
123            let mut next_level = Vec::new();
124            for children in current_level.chunks(NODE_SIZE) {
125                let mut child_vec = Vec::with_capacity(children.len());
126                let mut sizes = Vec::with_capacity(children.len());
127                let mut acc = 0;
128                for child in children {
129                    let node_size = child.size();
130                    acc += node_size;
131                    sizes.push(acc);
132                    child_vec.push(Some(child.clone()));
133                }
134                let branch = manager.allocate_node(Node::Branch {
135                    children: child_vec,
136                    sizes: Some(sizes),
137                });
138                next_level.push(branch);
139            }
140            current_level = next_level;
141            height += 1;
142        }
143
144        let root = current_level.into_iter().next().unwrap();
145        Self {
146            root,
147            size,
148            height,
149            manager,
150            cache: IndexCache::default(),
151            cache_policy: Box::new(crate::pvec::memory::AlwaysCache),
152            chunk_size,
153        }
154    }

185    /// Get the shift value for accessing elements at the current tree height.
186    #[inline(always)]
187    #[must_use]
188    pub fn shift(&self) -> usize {
189        self.height * NODE_BITS
190    }

192    /// Get a reference to the element at the specified index.
193    ///
194    /// Returns `None` if the index is out of bounds.
195    #[inline(always)]
196    pub fn get(&self, index: usize) -> Option<&T> {
197        if index >= self.size {
198            return None;
199        }
200        // NOTE: This immutable get does not use the cache for now.
201        self.root.get(index, self.shift())
202    }
203
204    /// Get a reference to the element at the specified index, using the index cache if policy allows.
205    /// Get a reference to the element at the specified index, using the index cache if policy allows.
206    pub fn get_with_cache(&mut self, index: usize) -> Option<&T> {
207        if !(self.cache_policy.should_cache(index)) {
208            return self.root.get(index, self.shift());
209        }
210        if index >= self.size {
211            return None;
212        }
213        // Check cache validity
214        if self.cache.is_valid() && self.cache.index == index && self.validate_cache_path() {
215            self.cache.record_hit();
216            self.root.get_by_path(
217                index,
218                self.shift(),
219                &self.cache.path[..self.cache.len],
220                &self.cache.ranges[..self.cache.len],
221            )
222        } else {
223            self.cache.record_miss();
224            let mut path = Vec::new();
225            let mut ranges = Vec::new();
226            let result = self
227                .root
228                .get_with_path(index, self.shift(), &mut path, &mut ranges);
229            self.cache.update(index, &path, &ranges);
230            result
231        }
232    }
233
234    fn validate_cache_path(&self) -> bool {
235        // Check if the cached path/ranges are valid in the current tree structure
236        let mut node = self.root.as_ref();
237        let mut shift = self.shift();
238        for (i, &child_index) in self.cache.path[..self.cache.len].iter().enumerate() {
239            match node {
240                Node::Branch { children, .. } => {
241                    if child_index >= children.len() {
242                        return false;
243                    }
244                    match &children[child_index] {
245                        Some(child) => {
246                            node = child.as_ref();
247                            shift = shift.saturating_sub(crate::pvec::node::NODE_BITS);
248                        },
249                        None => return false,
250                    }
251                },
252                Node::Leaf { elements } => {
253                    // Only the last path entry should access a leaf
254                    if i != self.cache.len - 1 {
255                        return false;
256                    }
257                    if child_index >= elements.inner().len() {
258                        return false;
259                    }
260                },
261            }
262        }
263        true
264    }

266    /// Update an element at the specified index, returning a new tree.
267    ///
268    /// Returns the original tree if the index is out of bounds.
269    #[inline(always)]
270    #[must_use]
271    pub fn update(&self, index: usize, value: T) -> Self {
272        if index >= self.size {
273            return self.clone();
274        }
275        if let Some(new_root) = self.root.update(&self.manager, index, value, self.shift()) {
276            Self {
277                root: new_root,
278                size: self.size,
279                height: self.height,
280                manager: self.manager.clone(),
281                cache: IndexCache::new(),
282                cache_policy: self.cache_policy.clone(),
283                chunk_size: self.chunk_size,
284            }
285        } else {
286            self.clone()
287        }
288    }

290    /// Splits the tree into two parts at the given index.
291    ///
292    /// Returns a tuple containing two trees: the first with elements from `0..index`,
293    /// and the second with elements from `index..len`.
294    #[inline(always)]
295    #[must_use]
296    pub fn split_at(&self, index: usize) -> (Self, Self) {
297        if index >= self.size {
298            return (self.clone(), Self::new());
299        }
300        if index == 0 {
301            return (Self::new(), self.clone());
302        }
303        let (left, right) = match self.height {
304            0 => {
305                // Handle split for leaf node directly
306                let chunk = match self.root.as_ref() {
307                    crate::pvec::node::Node::Leaf { elements } => elements,
308                    _ => panic!("Expected leaf node at height 0"),
309                };
310                let total = chunk.len();
311                if index > total {
312                    panic!("Split index {index} out of bounds for leaf of size {total}");
313                }
314                let left_chunk = self.manager.allocate_chunk({
315                    let mut c = crate::pvec::memory::Chunk::new_with_size(chunk.len());
316                    for i in 0..index {
317                        if let Some(val) = chunk.get(i) {
318                            c.push_back(val.clone());
319                        }
320                    }
321                    c
322                });
323                let right_chunk = self.manager.allocate_chunk({
324                    let mut c = crate::pvec::memory::Chunk::new_with_size(chunk.len());
325                    for i in index..total {
326                        if let Some(val) = chunk.get(i) {
327                            c.push_back(val.clone());
328                        }
329                    }
330                    c
331                });
332                let left_node = self.manager.allocate_node(crate::pvec::node::Node::Leaf {
333                    elements: left_chunk,
334                });
335                let right_node = self.manager.allocate_node(crate::pvec::node::Node::Leaf {
336                    elements: right_chunk,
337                });
338                (left_node, right_node)
339            },
340            _ => match self.root.split(index, self.shift(), &self.manager) {
341                Ok((l, r)) => (l, r),
342                Err(e) => panic!("Failed to split tree: {e}"),
343            },
344        };
345        (
346            Self {
347                root: left,
348                size: index,
349                height: self.height,
350                manager: self.manager.clone(),
351                cache: IndexCache::new(),
352                cache_policy: self.cache_policy.clone(),
353                chunk_size: self.chunk_size,
354            },
355            Self {
356                root: right,
357                size: self.size - index,
358                height: self.height,
359                manager: self.manager.clone(),
360                cache: IndexCache::new(),
361                cache_policy: self.cache_policy.clone(),
362                chunk_size: self.chunk_size,
363            },
364        )
365    }

367    /// Add an element to the end of the tree.
368    #[inline(always)]
369    #[must_use]
370    pub fn push_back(&self, value: T) -> Self {
371        let shift = self.shift();
372        let (new_root, split, overflow) =
373            self.root
374                .push_back(value, shift, self.chunk_size, &self.manager);
375
376        let mut result = self.clone();
377        result.size += 1;
378        result.cache.invalidate();
379
380        if split {
381            // Create a new root to handle the overflow
382            let mut children = Vec::with_capacity(2); // Only need space for 2 nodes
383            children.push(Some(new_root.clone()));
384
385            if let Some(overflow_node) = overflow.clone() {
386                children.push(Some(overflow_node));
387            }
388
389            // Create a minimal size table
390            let first_size = new_root.size();
391            let mut size_table = Vec::with_capacity(2);
392            size_table.push(first_size);
393
394            if let Some(ref overflow_node) = overflow {
395                size_table.push(first_size + overflow_node.size());
396            }
397
398            let new_branch = self.manager.allocate_node(Node::Branch {
399                children,
400                sizes: Some(size_table),
401            });
402
403            result.root = new_branch;
404            result.height += 1;
405        } else {
406            result.root = new_root;
407        }
408
409        result
410    }

412    /// Remove the last element from the tree.
413    ///
414    /// Returns a tuple containing the new tree and the removed element,
415    /// or `None` if the tree is empty.
416    #[inline(always)]
417    #[must_use]
418    pub fn pop_back(&self) -> Option<(Self, T)> {
419        if self.is_empty() {
420            return None;
421        }
422
423        let last_idx = self.size - 1;
424        let last_element = self.get(last_idx)?.clone();
425
426        // Create new tree without the last element
427        let (mut new_tree, _) = self.split_at(last_idx);
428        // If after pop the tree is empty, reset height to 0 and root to a new empty leaf
429        if new_tree.size == 0 {
430            let chunk = self
431                .manager
432                .allocate_chunk(crate::pvec::memory::Chunk::new_with_size(self.chunk_size));
433            let root = self
434                .manager
435                .allocate_node(crate::pvec::node::Node::leaf(chunk));
436            new_tree.root = root;
437            new_tree.height = 0;
438        }
439
440        Some((new_tree, last_element))
441    }

443    /// Collect all elements in the tree into a Vec
444    #[inline(always)]
445    #[must_use]
446    pub fn to_vec(&self) -> Vec<T> {
447        let mut out = Vec::with_capacity(self.size);
448        for i in 0..self.size {
449            if let Some(val) = self.get(i) {
450                out.push(val.clone());
451            }
452        }
453        out
454    }

458        (self.cache.hits, self.cache.misses)

462impl<T: PartialEq> PartialEq for Tree<T> {
463    #[inline(always)]
464    fn eq(&self, other: &Self) -> bool {
465        self.size == other.size && self.height == other.height && *self.root == *other.root
466    }
467}
