1//! # Transformation Utilities
2//!
3//! This module provides utilities for transforming functorial data types and building
4//! transformation pipelines. These tools help you work with functors in an ergonomic
5//! and composable way.
6//!
7//! ## Key Features
8//!
9//! - **Transformation Functions**: Apply transformations to functorial values
10//! - **Pipeline Builder**: Build fluent transformation chains with the `Pipeline` type
11//! - **Type-Safe Operations**: Leverage Rust's type system for safe transformations
12//!
13//! The utilities in this module work with any type that implements the `Functor` trait,
14//! including standard library types like `Option` and `Result`, and library types
15//! like `Either` and `Maybe`.
16
17use crate::prelude::Functor;
18
19// ===== Transformation Functions =====
20
21/// Applies a transformation to all functorial values in a collection.
22///
23/// This function maps a transformation over each functorial value in the collection,
24/// using the `Functor` trait's capabilities.
25///
26/// # Type Parameters
27///
28/// * `T`: A type that implements `Functor`
29/// * `F`: The transformation function type
30/// * `U`: The result type of the transformation
31///
32/// # Arguments
33///
34/// * `values`: The collection of functorial values to transform
35/// * `f`: The transformation function to apply to the content of each functor
36///
37/// # Returns
38///
39/// A new vector containing the transformed functorial values
40///
41/// # Examples
42///
43/// ```rust
44/// use rustica::utils::transform_utils::transform_all;
45/// use rustica::prelude::*;
46/// use rustica::datatypes::maybe::Maybe;
47///
48/// // Create a collection of Maybe values
49/// let values: Vec<Maybe<i32>> = vec![
50///     Maybe::Just(1),
51///     Maybe::Just(2),
52///     Maybe::Nothing,
53///     Maybe::Just(4)
54/// ];
55///
56/// // Transform the values inside each Maybe
57/// let doubles = transform_all(&values, |&x| x * 2);
58///
59/// // The transformation is applied only to the Just values
60/// assert_eq!(doubles[0], Maybe::Just(2));
61/// assert_eq!(doubles[1], Maybe::Just(4));
62/// assert_eq!(doubles[2], Maybe::Nothing);
63/// assert_eq!(doubles[3], Maybe::Just(8));
64/// ```
65///
66/// Works with standard library types too:
67///
68/// ```rust
69/// use rustica::utils::transform_utils::transform_all;
70/// use rustica::prelude::*;
71///
72/// // A collection of Option values
73/// let options: Vec<Option<String>> = vec![
74///     Some("hello".to_string()),
75///     None,
76///     Some("world".to_string())
77/// ];
78///
79/// // Transform the values inside each Option
80/// let uppercase = transform_all(&options, |s| s.to_uppercase());
81///
82/// assert_eq!(uppercase[0], Some("HELLO".to_string()));
83/// assert_eq!(uppercase[1], None);
84/// assert_eq!(uppercase[2], Some("WORLD".to_string()));
85/// ```
86#[inline]
87pub fn transform_all<T, F, U>(values: &[T], f: F) -> Vec<T::Output<U>>
88where
89    T: Functor,
90    F: Fn(&T::Source) -> U + Copy,
91    U: Clone,
92{
93    values.iter().map(|v| v.fmap(f)).collect()
94}
95
96/// Applies a transformation to a single optional value.
97///
98/// This function maps a transformation over the value inside an `Option` functor,
99/// preserving the `None` case and transforming the `Some` case.
100///
101/// # Type Parameters
102///
103/// * `T`: A type that implements `Functor`
104/// * `F`: The transformation function type
105/// * `U`: The result type of the transformation
106///
107/// # Arguments
108///
109/// * `value`: The optional functorial value to transform
110/// * `f`: The transformation function to apply to the content
111///
112/// # Returns
113///
114/// An optional transformed functorial value, or `None` if the input was `None`
115///
116/// # Examples
117///
118/// ```rust
119/// use rustica::utils::transform_utils::transform_chain;
120/// use rustica::prelude::*;
121/// use rustica::datatypes::maybe::Maybe;
122///
123/// // Apply a transformation to a Just value
124/// let maybe_just: Option<Maybe<i32>> = Some(Maybe::Just(5));
125/// let result1 = transform_chain(maybe_just, |&x| x * 2);
126/// assert_eq!(result1, Some(Maybe::Just(10)));
127///
128/// // Apply a transformation to a Nothing value
129/// let maybe_nothing: Option<Maybe<i32>> = Some(Maybe::Nothing);
130/// let result2 = transform_chain(maybe_nothing, |&x| x * 2);
131/// assert_eq!(result2, Some(Maybe::Nothing));
132///
133/// // When the outer Option is None, the result is None
134/// let none: Option<Maybe<i32>> = None;
135/// let result3 = transform_chain(none, |&x| x * 2);
136/// assert_eq!(result3, None);
137/// ```
138#[inline]
139pub fn transform_chain<T, F, U>(value: Option<T>, f: F) -> Option<T::Output<U>>
140where
141    T: Functor,
142    F: Fn(&T::Source) -> U,
143    U: Clone,
144{
145    value.map(|v| v.fmap(f))
146}

148// ===== Pipeline Builder =====
149
150/// A pipeline for building chains of transformations on functorial values.
151///
152/// This type provides a fluent interface for applying a sequence of transformations
153/// to a functorial value. It wraps the value and allows chaining transformations
154/// before finally extracting the result.
155///
156/// # Type Parameters
157///
158/// * `T`: The type of the functorial value in the pipeline
159///
160/// # Examples
161///
162/// ```rust
163/// use rustica::utils::transform_utils::Pipeline;
164/// use rustica::prelude::*;
165///
166/// // Create a pipeline with an Option value
167/// let pipeline = Pipeline::new(Some(5));
168///
169/// // Apply a series of transformations
170/// let result = pipeline
171///     .map(|&x| x * 2)      // Double the value
172///     .map(|x| x.to_string()) // Convert to string
173///     .extract();
174///
175/// assert_eq!(result, Some("10".to_string()));
176///
177/// // Works with Either as well
178/// use rustica::datatypes::either::Either;
179///
180/// let err_pipeline = Pipeline::new(Either::<&str, i32>::left("error"));
181/// let err_result = err_pipeline
182///     .map(|&x| x * 2)
183///     .extract();
184///
185/// assert_eq!(err_result, Either::left("error"));
186///
187/// let ok_pipeline = Pipeline::new(Either::<&str, i32>::right(7));
188/// let ok_result = ok_pipeline
189///     .map(|&x| x * 2)
190///     .extract();
191///
192/// assert_eq!(ok_result, Either::right(14));
193/// ```
194#[repr(transparent)]
195#[derive(Clone)]
196pub struct Pipeline<T>(T);
197
198impl<T> Pipeline<T> {
199    /// Creates a new pipeline with the given functorial value.
200    ///
201    /// # Arguments
202    ///
203    /// * `value`: The initial functorial value for the pipeline
204    ///
205    /// # Returns
206    ///
207    /// A new pipeline containing the value
208    ///
209    /// # Examples
210    ///
211    /// ```rust
212    /// use rustica::utils::transform_utils::Pipeline;
213    /// use rustica::datatypes::maybe::Maybe;
214    ///
215    /// // Create a pipeline with a Maybe value
216    /// let pipeline: Pipeline<Maybe<i32>> = Pipeline::new(Maybe::Just(42));
217    ///
218    /// // Create a pipeline with an Option value
219    /// let option_pipeline: Pipeline<Option<String>> = Pipeline::new(Some("hello".to_string()));
220    /// ```
221    #[inline]
222    pub fn new(value: T) -> Self {
223        Pipeline(value)
224    }
225
226    /// Extracts the final value from the pipeline.
227    ///
228    /// This method returns the transformed value after all pipeline operations
229    /// have been applied.
230    ///
231    /// # Returns
232    ///
233    /// The final transformed functorial value
234    ///
235    /// # Examples
236    ///
237    /// ```rust
238    /// use rustica::utils::transform_utils::Pipeline;
239    /// use rustica::prelude::*;
240    ///
241    /// // Create and extract from a pipeline
242    /// let result = Pipeline::new(Some(10))
243    ///     .extract();
244    ///
245    /// assert_eq!(result, Some(10));
246    /// ```
247    #[inline]
248    pub fn extract(self) -> T {
249        self.0
250    }
251}
252
253impl<T: Functor> Pipeline<T> {
254    /// Maps a transformation over the inner value, taking ownership.
255    ///
256    /// This method applies a transformation to the content of the functorial value
257    /// using the owned version of `fmap`. This is useful when you don't need to
258    /// preserve the original value.
259    ///
260    /// # Type Parameters
261    ///
262    /// * `F`: The transformation function type
263    /// * `U`: The result type of the transformation
264    ///
265    /// # Arguments
266    ///
267    /// * `f`: The transformation function to apply
268    ///
269    /// # Returns
270    ///
271    /// A new pipeline containing the transformed value
272    ///
273    /// # Examples
274    ///
275    /// ```rust
276    /// use rustica::utils::transform_utils::Pipeline;
277    /// use rustica::prelude::*;
278    ///
279    /// // Transform with map_owned
280    /// let result = Pipeline::new(Some(5))
281    ///     .map_owned(|x| x * 3)  // Using value by value
282    ///     .extract();
283    ///
284    /// assert_eq!(result, Some(15));
285    /// ```
286    #[inline]
287    pub fn map_owned<F, U>(self, f: F) -> Pipeline<T::Output<U>>
288    where
289        F: Fn(T::Source) -> U,
290        U: Clone,
291    {
292        Pipeline(self.0.fmap_owned(f))
293    }
294
295    /// Maps a transformation over the inner value, operating by reference.
296    ///
297    /// This method applies a transformation to the content of the functorial value
298    /// using the reference version of `fmap`. This preserves the pattern of
299    /// operating on values by reference.
300    ///
301    /// # Type Parameters
302    ///
303    /// * `F`: The transformation function type
304    /// * `U`: The result type of the transformation
305    ///
306    /// # Arguments
307    ///
308    /// * `f`: The transformation function to apply
309    ///
310    /// # Returns
311    ///
312    /// A new pipeline containing the transformed value
313    ///
314    /// # Examples
315    ///
316    /// ```rust
317    /// use rustica::utils::transform_utils::Pipeline;
318    /// use rustica::prelude::*;
319    /// use rustica::datatypes::either::Either;
320    ///
321    /// // Transform with map
322    /// let result = Pipeline::new(Either::<String, i32>::right(7))
323    ///     .map(|&x| x.to_string())  // Using value by reference
324    ///     .extract();
325    ///
326    /// assert_eq!(result, Either::<String, String>::right("7".to_string()));
327    /// ```
328    #[inline]
329    pub fn map<F, U>(self, f: F) -> Pipeline<T::Output<U>>
330    where
331        F: Fn(&T::Source) -> U,
332        U: Clone,
333    {
334        Pipeline(self.0.fmap(f))
335    }
336}
