1//! # Alternative Functors
2//!
3//! The `Alternative` trait represents applicative functors that also have a monoid structure.
4//! It extends the `Applicative` trait with operations for choice and failure.
5//!
6//! ## Core Operations
7//!
8//! - `empty_alt`: Provides an identity element (representing failure)
9//! - `alt`: Combines two alternatives (representing choice)
10//!
11//! ## Laws
12//!
13//! For a valid Alternative implementation, the following laws must hold:
14//!
15//! 1. Left identity: `empty_alt().alt(x) == x`
16//! 2. Right identity: `x.alt(empty_alt()) == x`
17//! 3. Associativity: `a.alt(b).alt(c) == a.alt(b.alt(c))`
18//! 4. Distributivity: `f.ap(a.alt(b)) == f.ap(a).alt(f.ap(b))`
19//! 5. Annihilation: `empty_alt().ap(a) == empty_alt()`
20//!
21//! ## Common Use Cases
22//!
23//! - Representing failure and recovery in computations
24//! - Parsing with multiple possible alternatives
25//! - Collecting multiple possible results
26//!
27//! ## Example
28//!
29//! ```rust
30//! use rustica::traits::alternative::Alternative;
31//!
32//! // Vec as Alternative
33//! let a = vec![1, 2];
34//! let b = vec![3, 4];
35//! let empty: Vec<i32> = <Vec<i32> as Alternative>::empty_alt::<i32>();
36//!
37//! // alt combines alternatives
38//! assert_eq!(a.alt(&b), vec![1, 2]);
39//! assert_eq!(empty.alt(&b), vec![3, 4]);
40//!
41//! // guard for conditional inclusion
42//! assert_eq!(Vec::<i32>::guard(true), vec![()]);
43//! assert_eq!(Vec::<i32>::guard(false), Vec::<()>::new());
44//!
45//! // many for repetition (Vec only)
46//! let xs = vec![42];
47//! assert_eq!(xs.many(), vec![vec![42]]);
48//! ```

51/// A trait for types that provide an alternative computation strategy.
52///
53/// `Alternative` extends `Applicative` with operations for choice and failure.
54/// It represents applicative functors that also have a monoid structure.
55///
56/// # Examples
57///
58/// ```rust
59/// use rustica::traits::alternative::Alternative;
60/// use rustica::traits::pure::Pure;
61///
62/// // Using Option as an Alternative
63/// let some_value: Option<i32> = Some(42);
64/// let none_value: Option<i32> = None;
65///
66/// // The alt method provides choice between alternatives
67/// assert_eq!(some_value.alt(&none_value), Some(42));
68/// assert_eq!(none_value.alt(&some_value), Some(42));
69///
70/// // The empty_alt method represents failure
71/// assert_eq!(Option::<i32>::empty_alt::<i32>(), None);
72///
73/// // Using guard for conditional computation
74/// assert_eq!(Option::<i32>::guard(true), Some(()));
75/// assert_eq!(Option::<i32>::guard(false), None);
76/// ```
77pub trait Alternative: Applicative {
78    /// Returns an empty value representing failure for the alternative computation.
79    ///
80    /// This is the identity element for the `alt` operation.
81    ///
82    /// # Examples
83    ///
84    /// ```rust
85    /// use rustica::traits::alternative::Alternative;
86    ///
87    /// assert_eq!(Option::<i32>::empty_alt::<i32>(), None);
88    /// assert_eq!(Vec::<i32>::empty_alt::<i32>(), Vec::new());
89    /// ```
90    fn empty_alt<T: Clone>() -> Self::Output<T>;
91
92    /// Combines two alternatives, choosing the first success.
93    ///
94    /// If `self` succeeds, it is returned. Otherwise, `other` is used.
95    ///
96    /// # Examples
97    ///
98    /// ```rust
99    /// use rustica::traits::alternative::Alternative;
100    ///
101    /// let a = Some(1);
102    /// let b = Some(2);
103    /// let c: Option<i32> = None;
104    ///
105    /// assert_eq!(a.alt(&b), Some(1));
106    /// assert_eq!(c.alt(&b), Some(2));
107    /// assert_eq!(c.alt(&c), None);
108    /// ```
109    fn alt(&self, other: &Self) -> Self
110    where
111        Self: Sized + Clone;
112
113    /// Returns a value if the condition is true, otherwise returns the empty value.
114    ///
115    /// # Laws
116    ///
117    /// - Identity: `Alternative::guard(true)` is not empty, `Alternative::guard(false)` is empty.
118    ///
119    /// # Examples
120    /// ```rust
121    /// use rustica::traits::alternative::Alternative;
122    /// assert_eq!(Vec::<i32>::guard(true), vec![()]);
123    /// assert_eq!(Vec::<i32>::guard(false), Vec::<()>::new());
124    /// assert_eq!(Option::<i32>::guard(true), Some(()));
125    /// assert_eq!(Option::<i32>::guard(false), None);
126    /// ```
127    fn guard(condition: bool) -> Self::Output<()>;
128
129    /// Repeats the structure zero or more times, collecting the results.
130    ///
131    /// # Laws
132    ///
133    /// - many(empty) == empty
134    /// - many(x) for non-empty x yields a collection containing x
135    ///
136    /// # Examples
137    /// ```rust
138    /// use rustica::traits::alternative::Alternative;
139    /// let empty: Vec<i32> = Vec::new();
140    /// assert_eq!(empty.many(), Vec::<Vec<i32>>::new());
141    /// let xs = vec![1,2];
142    /// assert_eq!(xs.many(), vec![vec![1,2]]);
143    ///
144    /// let none: Option<i32> = None;
145    /// let some = Some(42);
146    /// assert_eq!(none.many(), None);
147    /// assert_eq!(some.many(), Some(vec![42]));
148    /// ```
149    fn many(&self) -> Self::Output<Vec<Self::Source>>
150    where
151        Self::Source: Clone;
152}
153
154impl<T> Alternative for Option<T>
155where
156    T: Clone,
157{
158    fn empty_alt<U>() -> Self::Output<U> {
159        None
160    }
161
162    fn alt(&self, other: &Self) -> Self {
163        self.clone().or_else(|| other.clone())
164    }
165
166    fn guard(condition: bool) -> Self::Output<()> {
167        condition.then_some(())
168    }
169
170    fn many(&self) -> Self::Output<Vec<Self::Source>>
171    where
172        Self::Source: Clone,
173    {
174        self.as_ref().map(|value| vec![value.clone()])
175    }
176}
177
178impl<T> Alternative for Vec<T>
179where
180    T: Clone,
181{
182    fn empty_alt<U>() -> Self::Output<U> {
183        Vec::new()
184    }
185
186    fn alt(&self, other: &Self) -> Self {
187        if self.is_empty() {
188            other.clone()
189        } else {
190            self.clone()
191        }
192    }
193
194    fn guard(condition: bool) -> Self::Output<()> {
195        if condition {
196            vec![()]
197        } else {
198            Vec::new()
199        }
200    }
201
202    fn many(&self) -> Self::Output<Vec<Self::Source>>
203    where
204        Self::Source: Clone,
205    {
206        if self.is_empty() {
207            Vec::new()
208        } else {
209            vec![self.clone()]
210        }
211    }
212}
