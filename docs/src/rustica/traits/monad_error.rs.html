1//! # MonadError
2//!
3//! The `MonadError` module provides trait definitions for error handling in a monadic context.
4//! It extends the `Monad` trait with error-specific operations.
5//!
6//! MonadError is particularly useful for implementing error handling strategies that follow
7//! functional programming principles. It allows for catching, handling, and throwing errors
8//! within a monadic context without breaking the computation chain.
9//!
10//! # Relationship to other traits
11//!
12//! MonadError is an extension of the Monad trait, providing specialized operations for error handling:
13//!
14//! ```text
15//! Functor -> Applicative -> Monad -> MonadError
16//! ```
17//!
18//! # Mathematical Definition
19//!
20//! A MonadError is a monad with additional error handling structure:
21//! - `throw`: E -> M A
22//! - `catch`: M A -> (E -> M A) -> M A
23//!
24//! # Laws
25//!
26//! For a valid MonadError implementation, the following laws must hold:
27//!
28//! 1. Left Catch Law:
29//!    ```text
30//!    throw(e).catch(h) == h(e)
31//!    ```
32//!    Catching an error that was just thrown should be equivalent to just handling that error.
33//!
34//! 2. Right Catch Law:
35//!    ```text
36//!    m.catch(throw) == m
37//!    ```
38//!    Catching with throw as the handler should be a no-op.
39//!
40//! 3. Associativity Catch Law:
41//!    ```text
42//!    m.catch(h1).catch(h2) == m.catch(e -> h1(e).catch(h2))
43//!    ```
44//!    Nested catches can be rewritten as a single catch with a composed handler.
45//!
46//! # Examples
47//!
48//! ```rust
49//! use rustica::traits::monad_error::{MonadError, ErrorMapper};
50//! use rustica::traits::monad::Monad;
51//! use rustica::traits::pure::Pure;
52//! use rustica::traits::functor::Functor;
53//!
54//! // Define a simple custom error that implements Clone
55//! #[derive(Debug, Clone, PartialEq)]
56//! struct AppError {
57//!     message: String,
58//!     code: i32,
59//! }
60//!
61//! // Using Result as a MonadError to handle errors
62//! let success_result: Result<i32, AppError> = Result::<i32, AppError>::pure(&42);
63//! let error_result: Result<i32, AppError> = Result::<i32, AppError>::throw(AppError {
64//!     message: "Item not found".to_string(),
65//!     code: 404,
66//! });
67//!
68//! // Catching errors
69//! let handled = error_result.catch(|e| {
70//!     if e.code == 404 {
71//!         Ok(0)  // Default value for not found
72//!     } else {
73//!         Err(e.clone()) // Pass through other errors
74//!     }
75//! });
76//!
77//! assert_eq!(handled, Ok(0));
78//!
79//! // Transforming between error types using ErrorMapper
80//! let string_error = success_result.map_error_to::<String, _>(|e| format!("Error {}: {}", e.code, e.message));
81//!
82//! assert_eq!(string_error, Ok(42));
83//! ```

87/// A trait for monads that can handle errors, extending the basic Monad trait.
88///
89/// MonadError provides operations for throwing errors and catching errors within
90/// a monadic context. It allows for robust error handling while maintaining the
91/// benefits of monadic computation chains.
92///
93/// # Type Parameters
94/// * `E`: The error type that can be thrown and caught
95///
96/// # Laws
97/// For a valid MonadError implementation, the following laws must hold:
98///
99/// 1. Left Catch Law:
100///    throw(e).catch(h) == h(e)
101///    Catching an error that was just thrown should be equivalent to just handling that error.
102///
103/// 2. Right Catch Law:
104///    m.catch(throw) == m
105///    Catching with throw as the handler should be a no-op.
106///
107/// 3. Associativity Catch Law:
108///    m.catch(h1).catch(h2) == m.catch(e -> h1(e).catch(h2))
109///    Nested catches can be rewritten as a single catch with a composed handler.
110pub trait MonadError<E>: Monad
111where
112    E: Clone + Debug,
113{
114    /// Creates a new instance in an error state.
115    ///
116    /// This is the equivalent of throwing an exception in languages with exceptions.
117    ///
118    /// # Type Parameters
119    /// * `T`: The type of value that would be contained in a successful result
120    ///
121    /// # Parameters
122    /// * `error`: The error value to throw
123    ///
124    /// # Returns
125    /// A new monadic value in an error state
126    fn throw<T: Clone>(error: E) -> Self::Output<T>;
127
128    /// Handles an error by applying a function that can recover from the error.
129    ///
130    /// If this monadic value is in an error state, applies the given function to
131    /// recover. Otherwise, returns the current successful value.
132    ///
133    /// # Type Parameters
134    /// * `F`: The type of the error-handling function
135    ///
136    /// # Parameters
137    /// * `f`: A function that takes an error and returns a new monadic value
138    ///
139    /// # Returns
140    /// Either the original successful value or the result of applying the
141    /// recovery function to the error
142    fn catch<F>(&self, f: F) -> Self::Output<Self::Source>
143    where
144        F: Fn(&E) -> Self::Output<Self::Source>,
145        Self::Source: Clone;
146
147    /// Handles an error by applying a function that can recover from the error, consuming self.
148    ///
149    /// This variant of `catch` takes ownership of `self`, allowing for more efficient
150    /// implementations when the original monad is no longer needed.
151    ///
152    /// # Type Parameters
153    /// * `F`: The type of the error-handling function
154    ///
155    /// # Parameters
156    /// * `f`: A function that takes an error and returns a new monadic value
157    ///
158    /// # Returns
159    /// Either the original successful value or the result of applying the
160    /// recovery function to the error
161    fn catch_owned<F>(self, f: F) -> Self::Output<Self::Source>
162    where
163        F: Fn(E) -> Self::Output<Self::Source>,
164        Self::Source: Clone,
165        Self: Sized;
166}

168/// A trait for types that can map their error type to a different error type.
169///
170/// This trait is separate from MonadError to allow for more flexible error handling
171/// where the error type can be transformed.
172///
173/// # Type Parameters
174/// * `E`: The original error type
175///
176/// # Examples
177///
178/// ```rust
179/// use rustica::traits::monad_error::ErrorMapper;
180///
181/// // Use a simple String as our error type, which implements Clone
182/// let result: Result<i32, String> = Ok(42);
183/// let string_result = result.map_error_to::<String, _>(|e| format!("Error: {}", e));
184/// assert_eq!(string_result, Ok(42));
185/// ```
186pub trait ErrorMapper<E>
187where
188    E: Clone + Debug,
189{
190    /// The source type contained in the monad
191    type Source;
192
193    /// Transforms the error type using the given function.
194    ///
195    /// This allows for adapting between different error types while preserving
196    /// the successful value.
197    ///
198    /// # Type Parameters
199    /// * `NewE`: The new error type
200    /// * `F`: The type of the error-mapping function
201    ///
202    /// # Parameters
203    /// * `f`: A function that transforms the error type
204    ///
205    /// # Returns
206    /// A new monadic value with the same success type but a different error type
207    fn map_error_to<NewE, F>(&self, f: F) -> Result<Self::Source, NewE>
208    where
209        F: Fn(&E) -> NewE,
210        NewE: Clone + Debug,
211        Self::Source: Clone;
212
213    /// Transforms the error type using the given function, consuming self.
214    ///
215    /// This variant of `map_error_to` takes ownership of `self`, allowing for more
216    /// efficient implementations when the original monad is no longer needed.
217    ///
218    /// # Type Parameters
219    /// * `NewE`: The new error type
220    /// * `F`: The type of the error-mapping function
221    ///
222    /// # Parameters
223    /// * `f`: A function that transforms the error type
224    ///
225    /// # Returns
226    /// A new monadic value with the same success type but a different error type
227    fn map_error_to_owned<NewE, F>(self, f: F) -> Result<Self::Source, NewE>
228    where
229        F: Fn(E) -> NewE,
230        NewE: Clone + Debug,
231        Self::Source: Clone,
232        Self: Sized;
233}

235// Implementation for Result
236impl<T: Clone, E: Clone + Debug> MonadError<E> for Result<T, E> {
237    #[inline]
238    fn throw<U: Clone>(error: E) -> Self::Output<U> {
239        Err(error)
240    }
241
242    #[inline]
243    fn catch<F>(&self, f: F) -> Self::Output<Self::Source>
244    where
245        F: Fn(&E) -> Self::Output<Self::Source>,
246        Self::Source: Clone,
247    {
248        match self {
249            Ok(value) => Ok(value.clone()),
250            Err(error) => f(error),
251        }
252    }
253
254    #[inline]
255    fn catch_owned<F>(self, f: F) -> Self::Output<Self::Source>
256    where
257        F: Fn(E) -> Self::Output<Self::Source>,
258        Self::Source: Clone,
259        Self: Sized,
260    {
261        match self {
262            Ok(value) => Ok(value),
263            Err(error) => f(error),
264        }
265    }
266}

300// Add implementations for Option, treating None as an error
301// Option doesn't have an explicit error type, so we use () as a placeholder
302impl<T: Clone> MonadError<()> for Option<T> {
