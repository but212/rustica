1//! Prelude: Utilities & Extension Traits
2//!
3//! This module re-exports various utility functions and extension traits from Rustica to make them available in a single import.
4//!
5//! # Key Utilities
6//!
7//! - `error_utils`: Error handling and conversion helpers (e.g., `collect_results`, `flatten_errors`)
8//! - `hkt_utils`: Higher-kinded type utilities (e.g., HKT conversions, type witnesses)
9//! - `transform_utils`: Function composition, transformation, and chaining utilities
10//!
11//! # Usage Examples
12//!
13//! ```rust
14//! use rustica::prelude::utils::*;
15//!
16//! // --- Error Utilities ---
17//! let results = vec![Ok(1), Ok(2), Ok(3)];
18//! let ok: Result<Vec<i32>, &str> = sequence(results);
19//! assert_eq!(ok, Ok(vec![1, 2, 3]));
20//!
21//! use rustica::datatypes::validated::Validated;
22//! let inputs = vec!["1", "not_a_number", "3"];
23//! let parsed: Validated<String, Vec<i32>> =
24//!     traverse_validated(inputs, |s| s.parse::<i32>().map_err(|_| format!("bad: {}", s)));
25//! assert!(parsed.is_invalid());
26//! assert_eq!(parsed.errors().len(), 1);
27//! assert!(parsed.errors()[0].contains("bad: not_a_number"));
28//!
29//! use rustica::datatypes::either::Either;
30//! let ok_result: Result<i32, &str> = Ok(42);
31//! let either: Either<&str, i32> = result_to_either(ok_result);
32//! assert_eq!(either, Either::right(42));
33//!
34//! // --- HKT Utilities ---
35//! let numbers = vec![1, 2, 3, 4, 5, 6];
36//! let evens_squared = filter_map(numbers, |&n| n % 2 == 0, |n| n * n);
37//! assert_eq!(evens_squared, vec![4, 16, 36]);
38//!
39//! let a = vec![1, 2, 3];
40//! let b = vec![4, 5, 6];
41//! let summed = zip_with(a, b, |x, y| x + y);
42//! assert_eq!(summed, vec![5, 7, 9]);
43//!
44//! let double_if_even = |n: i32| if n % 2 == 0 { Some(n * 2) } else { None };
45//! let result = pipeline_option(10, vec![double_if_even]);
46//! assert_eq!(result, Some(20));
47//!
48//! // --- Transform Utilities ---
49//! use rustica::datatypes::maybe::Maybe;
50//! let maybes = vec![Maybe::Just(1), Maybe::Just(2), Maybe::Nothing];
51//! let doubled: Vec<Maybe<i32>> = transform_all(&maybes, |x| x * 2);
52//! assert_eq!(doubled, vec![Maybe::Just(2), Maybe::Just(4), Maybe::Nothing]);
53//!
54//! use rustica::utils::transform_utils::Pipeline;
55//! let result = Pipeline::new(Some(5))
56//!     .map(|&x| x * 3)
57//!     .map(|x| x.to_string())
58//!     .extract();
59//! assert_eq!(result, Some("15".to_string()));
60//! ```
61//!
62//! # Note
63//!
64//! These utilities form powerful combinations when used with Rustica's datatypes, traits, and transformers.
