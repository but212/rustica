1//! # Higher-Kinded Type Utilities
2//!
3//! This module provides utility functions and combinators for working with higher-kinded types
4//! and functional programming patterns. These utilities simplify common operations and enable
5//! more expressive functional code in Rust.
6//!
7//! ## Categories of Utilities
8//!
9//! - **Collection Operations**: Functions like `filter_map` and `zip_with` that work with collections
10//! - **Pipeline Functions**: Utilities like `pipeline_option` and `pipeline_result` for chaining operations
11//! - **Function Transformers**: Utilities that transform functions to work with different contexts
12//! - **Composition Utilities**: Functions for combining and composing operations
13//!
14//! These utilities complement the traits in the `traits` module and provide ready-to-use
15//! implementations of common functional programming patterns.
16
17// ===== Collection Operations =====

19/// Filters and maps a collection in a single pass.
20///
21/// This function combines filtering and mapping operations for better performance
22/// and cleaner code. It first filters elements using the predicate and then
23/// applies the transformation function to the filtered elements.
24///
25/// # Type Parameters
26///
27/// * `A`: The type of elements in the input collection
28/// * `B`: The type of elements in the output collection
29/// * `C`: The type of the input collection
30/// * `P`: The type of the predicate function
31/// * `F`: The type of the mapping function
32///
33/// # Arguments
34///
35/// * `collection`: The input collection to filter and map
36/// * `predicate`: A function that decides which elements to keep
37/// * `f`: A function that transforms kept elements
38///
39/// # Returns
40///
41/// A new vector containing the transformed elements that passed the predicate
42///
43/// # Examples
44///
45/// ```rust
46/// use rustica::utils::hkt_utils::filter_map;
47///
48/// let numbers = vec![1, 2, 3, 4, 5, 6];
49///
50/// // Keep only even numbers and square them
51/// let result = filter_map(
52///     numbers,
53///     |&n| n % 2 == 0,  // Keep only even numbers
54///     |n| n * n        // Square the kept numbers
55/// );
56///
57/// assert_eq!(result, vec![4, 16, 36]);
58/// ```
59#[inline]
60pub fn filter_map<A, B, C, P, F>(collection: C, predicate: P, f: F) -> Vec<B>
61where
62    C: IntoIterator<Item = A>,
63    P: Fn(&A) -> bool,
64    F: Fn(A) -> B,
65{
66    collection.into_iter().filter(predicate).map(f).collect()
67}

69/// Combines elements from two collections using a combining function.
70///
71/// This function pairs elements from two collections and applies a combining function
72/// to each pair, producing a new collection of the results. The length of the output
73/// will be the minimum of the lengths of the two input collections.
74///
75/// # Type Parameters
76///
77/// * `A`: The type of elements in the first input collection
78/// * `B`: The type of elements in the second input collection
79/// * `C`: The type of elements in the output collection
80/// * `F`: The type of the combining function
81///
82/// # Arguments
83///
84/// * `xs`: The first input collection
85/// * `ys`: The second input collection
86/// * `f`: A function that combines elements from both collections
87///
88/// # Returns
89///
90/// A new vector containing the combined elements
91///
92/// # Examples
93///
94/// ```rust
95/// use rustica::utils::hkt_utils::zip_with;
96///
97/// let numbers = vec![1, 2, 3, 4];
98/// let multipliers = vec![10, 20, 30];
99///
100/// // Multiply corresponding elements
101/// let result = zip_with(
102///     numbers,
103///     multipliers,
104///     |n, m| n * m
105/// );
106///
107/// assert_eq!(result, vec![10, 40, 90]);  // Note: only 3 elements (minimum length)
108/// ```
109#[inline]
110pub fn zip_with<A, B, C, F>(xs: Vec<A>, ys: Vec<B>, f: F) -> Vec<C>
111where
112    F: Fn(A, B) -> C,
113{
114    xs.into_iter().zip(ys).map(|(x, y)| f(x, y)).collect()
115}

119/// Chains a sequence of operations that may return `Option<T>`.
120///
121/// This function allows you to chain multiple operations where each operation
122/// may fail (returning `None`). The chain short-circuits on the first `None`.
123///
124/// # Type Parameters
125///
126/// * `A`: The type of the initial value
127/// * `B`: The type that the initial value is converted to and operations work with
128/// * `Func`: The type of the operation functions
129///
130/// # Arguments
131///
132/// * `initial`: The starting value for the pipeline
133/// * `operations`: A sequence of operations to apply
134///
135/// # Returns
136///
137/// The final transformed value, or `None` if any operation failed
138///
139/// # Examples
140///
141/// ```rust
142/// use rustica::utils::hkt_utils::pipeline_option;
143///
144/// // Define operations that work with our pipeline
145/// let parse_int = |s: &str| -> Option<i32> { s.parse::<i32>().ok() };
146///
147/// // Create operations for our pipeline
148/// let double_if_even = |n: i32| -> Option<i32> {
149///     if n % 2 == 0 { Some(n * 2) } else { None }
150/// };
151///
152/// // Chain operations - this will succeed
153/// let result1 = pipeline_option(
154///     10,  // Initial i32 value
155///     vec![double_if_even]
156/// );
157/// assert_eq!(result1, Some(20));  // 10 -> Some(20)
158///
159/// // Chain operations - this will fail
160/// let result2 = pipeline_option(
161///     11,  // Initial i32 value (odd number)
162///     vec![double_if_even]
163/// );
164/// assert_eq!(result2, None);
165/// ```
166#[inline]
167pub fn pipeline_option<A, B, Func>(initial: A, operations: Vec<Func>) -> Option<B>
168where
169    Func: Fn(B) -> Option<B>,
170    A: Into<B>,
171{
172    let mut iter = operations.into_iter();
173
174    // Handle the first operation or just use the initial value
175    let initial_result = match iter.next() {
176        Some(first_op) => first_op(initial.into()),
177        None => Some(initial.into()),
178    };
179
180    // Apply the remaining operations
181    iter.fold(initial_result, |acc, op| match acc {
182        Some(value) => op(value),
183        None => None,
184    })
185}

187/// Chains a sequence of operations that may return `Result<T, E>`.
188///
189/// This function allows you to chain multiple operations where each operation
190/// may fail (returning `Err`). The chain short-circuits on the first `Err`.
191///
192/// # Type Parameters
193///
194/// * `A`: The type of the initial value
195/// * `B`: The type that the initial value is converted to and operations work with
196/// * `E`: The error type
197/// * `Func`: The type of the operation functions
198///
199/// # Arguments
200///
201/// * `initial`: The starting value for the pipeline
202/// * `operations`: A sequence of operations to apply
203///
204/// # Returns
205///
206/// The final transformed value, or the first error encountered
207///
208/// # Examples
209///
210/// ```rust
211/// use rustica::utils::hkt_utils::pipeline_result;
212///
213/// // Some example operations that might fail
214/// let checked_div = |n: i32| -> Result<i32, &'static str> {
215///     if n % 3 == 0 {
216///         Ok(n / 3)
217///     } else {
218///         Err("not divisible by 3")
219///     }
220/// };
221///
222/// let checked_add = |n: i32| -> Result<i32, &'static str> {
223///     if n < 100 {
224///         Ok(n + 10)
225///     } else {
226///         Err("result too large")
227///     }
228/// };
229///
230/// // Chain operations - this will succeed
231/// let result1 = pipeline_result(
232///     9,  // Initial value
233///     vec![checked_div, checked_add]
234/// );
235/// assert_eq!(result1, Ok(13));  // (9 / 3) + 10 = 13
236///
237/// // Chain operations - this will fail at the first operation
238/// let result2 = pipeline_result(
239///     10,  // Initial value (not divisible by 3)
240///     vec![checked_div, checked_add]
241/// );
242/// assert_eq!(result2, Err("not divisible by 3"));
243/// ```
244#[inline]
245pub fn pipeline_result<A, B, E, Func>(initial: A, operations: Vec<Func>) -> Result<B, E>
246where
247    Func: Fn(B) -> Result<B, E>,
248    A: Into<B>,
249{
250    let mut iter = operations.into_iter();
251
252    // Handle the first operation or just use the initial value
253    let initial_result = match iter.next() {
254        Some(first_op) => first_op(initial.into()),
255        None => Ok(initial.into()),
256    };
257
258    // Apply the remaining operations
259    iter.fold(initial_result, |acc, op| match acc {
260        Ok(value) => op(value),
261        Err(e) => Err(e),
262    })
263}

265/// Convenience wrapper around `pipeline_result` with a more intuitive name.
266///
267/// This function is identical to `pipeline_result` but with a name that better
268/// reflects its error handling behavior.
269///
270/// # Type Parameters
271///
272/// * `A`: The type of the initial value
273/// * `B`: The type that the initial value is converted to and operations work with
274/// * `E`: The error type
275/// * `F`: The type of the operation functions
276///
277/// # Arguments
278///
279/// * `initial`: The starting value for the pipeline
280/// * `operations`: A sequence of operations to apply
281///
282/// # Returns
283///
284/// The final transformed value, or the first error encountered
285///
286/// # Examples
287///
288/// ```rust
289/// use rustica::utils::hkt_utils::try_pipeline;
290///
291/// // Define operations that may fail
292/// let parse_as_f64 = |s: &str| -> Result<f64, &'static str> {
293///     s.parse::<f64>().map_err(|_| "parse error")
294/// };
295///
296/// let sqrt = |n: f64| -> Result<f64, &'static str> {
297///     if n >= 0.0 {
298///         Ok(n.sqrt())
299///     } else {
300///         Err("cannot take square root of negative number")
301///     }
302/// };
303///
304/// // Chain operations with try_pipeline
305/// let result = try_pipeline(
306///     "16".to_string(),
307///     vec![
308///         Box::new(|s: String| parse_as_f64(&s).map(|n| n.to_string())) as Box<dyn Fn(String) -> Result<String, &'static str>>,
309///         Box::new(|s: String| parse_as_f64(&s).and_then(sqrt).map(|n| n.to_string()))
310///     ]
311/// );
312///
313/// assert_eq!(result.map(|s| s.parse::<f64>().unwrap()), Ok(4.0));
314/// ```
315#[inline]
316pub fn try_pipeline<A, B, E, F>(initial: A, operations: Vec<F>) -> Result<B, E>
317where
318    F: Fn(B) -> Result<B, E>,
319    A: Into<B>,
320{
321    pipeline_result(initial, operations)
322}

326/// Lifts a function to work with `Option` values.
327///
328/// This function transforms a regular function into one that works with
329/// `Option` values by applying the function only when the option is `Some`.
330///
331/// # Type Parameters
332///
333/// * `A`: The input type of the original function
334/// * `B`: The output type of the original function
335/// * `F`: The type of the function to lift
336///
337/// # Arguments
338///
339/// * `f`: The function to lift to work with `Option`
340///
341/// # Returns
342///
343/// A new function that takes an `Option<A>` and returns an `Option<B>`
344///
345/// # Examples
346///
347/// ```rust
348/// use rustica::utils::hkt_utils::lift_option;
349///
350/// // Define a regular function
351/// let double = |x: i32| x * 2;
352///
353/// // Lift it to work with Option
354/// let option_double = lift_option(double);
355///
356/// // Use the lifted function
357/// assert_eq!(option_double(Some(5)), Some(10));
358/// assert_eq!(option_double(None), None);
359///
360/// // We can also use it in a pipeline
361/// let values: Vec<Option<i32>> = vec![Some(1), None, Some(3)];
362/// let doubled: Vec<Option<i32>> = values.into_iter().map(option_double).collect();
363/// assert_eq!(doubled, vec![Some(2), None, Some(6)]);
364/// ```
365#[inline]
366pub fn lift_option<A, B, F>(f: F) -> impl Fn(Option<A>) -> Option<B>
367where
368    F: Fn(A) -> B,
369{
370    move |opt| opt.map(&f)
371}

373/// Maps a function over a `Result` value.
374///
375/// This function applies a transformation to the success value of a `Result`,
376/// leaving any error value unchanged.
377///
378/// # Type Parameters
379///
380/// * `A`: The success type of the input `Result`
381/// * `B`: The success type of the output `Result`
382/// * `E`: The error type of both input and output `Result`
383/// * `F`: The type of the mapping function
384///
385/// # Arguments
386///
387/// * `result`: The input `Result` value
388/// * `f`: The function to apply to the success value
389///
390/// # Returns
391///
392/// A new `Result` with the success value transformed
393///
394/// # Examples
395///
396/// ```rust
397/// use rustica::utils::hkt_utils::map_result;
398///
399/// // Example Result values
400/// let ok_value: Result<i32, &str> = Ok(5);
401/// let err_value: Result<i32, &str> = Err("error occurred");
402///
403/// // Transform the success value
404/// let mapped_ok = map_result(ok_value, |n| n * 2);
405/// assert_eq!(mapped_ok, Ok(10));
406///
407/// // Error values pass through unchanged
408/// let mapped_err = map_result(err_value, |n| n * 2);
409/// assert_eq!(mapped_err, Err("error occurred"));
410/// ```
411#[inline]
412pub fn map_result<A, B, E, F>(result: Result<A, E>, f: F) -> Result<B, E>
413where
414    F: Fn(A) -> B,
415{
416    result.map(f)
417}

421/// Applies multiple operations to a single input value.
422///
423/// This function takes a value and a collection of operations, applies each
424/// operation to the input value, and collects the results. This is useful when
425/// you need to transform a single input in multiple different ways.
426///
427/// # Type Parameters
428///
429/// * `A`: The type of the input value
430/// * `B`: The type of the output values
431/// * `F`: The type of the operation functions
432///
433/// # Arguments
434///
435/// * `input`: The input value to transform
436/// * `operations`: A collection of operations to apply
437///
438/// # Returns
439///
440/// A vector containing the results of applying each operation to the input
441///
442/// # Examples
443///
444/// ```rust
445/// use rustica::utils::hkt_utils::fan_out;
446///
447/// // Single input value
448/// let value = 10;
449///
450/// // Multiple operations to apply
451/// let ops = vec![
452///     |n: i32| n + 5,
453///     |n: i32| n * 2,
454///     |n: i32| n * n
455/// ];
456///
457/// // Apply all operations to the input
458/// let results = fan_out(value, ops);
459///
460/// // We get the result of each operation
461/// assert_eq!(results, vec![15, 20, 100]);
462/// ```
463#[inline]
464pub fn fan_out<A: Clone, B, F>(input: A, operations: Vec<F>) -> Vec<B>
465where
466    F: Fn(A) -> B,
467{
468    operations.into_iter().map(|op| op(input.clone())).collect()
469}

471/// Composes multiple transformations into a single function.
472///
473/// This function takes a sequence of transformations and composes them into
474/// a single function that applies all transformations in sequence.
475///
476/// # Type Parameters
477///
478/// * `A`: The input and intermediate type
479/// * `B`: The final output type
480///
481/// # Arguments
482///
483/// * `transformations`: A vector of transformation functions
484///
485/// # Returns
486///
487/// A boxed function that applies all transformations in sequence
488///
489/// # Examples
490///
491/// ```rust
492/// use rustica::utils::hkt_utils::compose_all;
493/// use std::ops::Add;
494///
495/// // Define some transformations
496/// let add_5: Box<dyn Fn(i32) -> i32> = Box::new(|n| n + 5);
497/// let multiply_2: Box<dyn Fn(i32) -> i32> = Box::new(|n| n * 2);
498/// let square: Box<dyn Fn(i32) -> i32> = Box::new(|n| n * n);
499///
500/// // Compose all transformations
501/// let composed: Box<dyn Fn(i32) -> i32> = compose_all(vec![add_5, multiply_2, square]);
502///
503/// // ((10 + 5) * 2)² = 30² = 900
504/// assert_eq!(composed(10), 900);
505/// ```
506#[inline]
507pub fn compose_all<A, B>(transformations: Vec<Box<dyn Fn(A) -> A>>) -> Box<dyn Fn(A) -> B>
508where
509    A: 'static,
510    B: 'static + From<A>,
511{
512    Box::new(move |input| {
513        let result = transformations.iter().fold(input, |acc, f| f(acc));
514        B::from(result)
515    })
