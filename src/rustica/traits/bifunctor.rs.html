1//! # Bifunctors
2//!
3//! A bifunctor is a type constructor that takes two type arguments and is a functor in both arguments.
4//! It extends the concept of a functor to data types with two type parameters, allowing functions
5//! to be mapped over both parameters independently or simultaneously.
6//!
7//! ## Mathematical Definition
8//!
9//! In category theory, a bifunctor is a functor of two arguments:
10//!
11//! ```text
12//! F: C × D → E
13//! ```
14//!
15//! Where C, D, and E are categories. In Rust terms, a bifunctor provides three operations:
16//!
17//! - `first`: Map a function over the first type parameter
18//! - `second`: Map a function over the second type parameter
19//! - `bimap`: Map two functions over both type parameters simultaneously
20//!
21//! ## Laws
22//!
23//! For a valid bifunctor implementation, the following laws must hold:
24//!
25//! 1. Identity:
26//! ```text
27//! bimap(id, id) == id
28//! ```
29//!
30//! 2. Composition:
31//! ```text
32//! bimap(f . g, h . i) == bimap(f, h) . bimap(g, i)
33//! ```
34//!
35//! ## Common Bifunctors
36//!
37//! Some common bifunctors in Rust include:
38//!
39//! - `Result<T, E>`: Functor in both the success type `T` and error type `E`
40//! - `Either<L, R>`: Functor in both left type `L` and right type `R`
41//! - `Tuple2<A, B>`: Functor in both tuple components `A` and `B`
42//!
43//! ## Use Cases
44//!
45//! Bifunctors are particularly useful for:
46//!
47//! 1. **Error Handling**: Transform both success and error types in `Result`
48//! 2. **Data Transformation**: Process pairs of values independently
49//! 3. **Type Conversion**: Convert between different type combinations while preserving structure
50//!
51//! ## Examples
52//!
53//! ```rust
54//! use rustica::traits::bifunctor::Bifunctor;
55//! use rustica::traits::hkt::{HKT, BinaryHKT};
56//!
57//! // Example with a tuple type
58//! #[derive(Debug, PartialEq, Clone)]
59//! struct BiTuple<A, B>(A, B);
60//!
61//! impl<A, B> HKT for BiTuple<A, B> {
62//!     type Source = A;
63//!     type Output<U> = BiTuple<U, B>;
64//! }
65//!
66//! impl<A, B> BinaryHKT for BiTuple<A, B> {
67//!     type Source2 = B;
68//!     type BinaryOutput<U, V> = BiTuple<U, V>;
69//!
70//!     fn map_second<F, NewType2>(&self, f: F) -> Self::BinaryOutput<Self::Source, NewType2>
71//!     where
72//!         F: Fn(&Self::Source2) -> NewType2,
73//!         Self::Source: Clone,
74//!     {
75//!         BiTuple(self.0.clone(), f(&self.1))
76//!     }
77//!
78//!     fn map_second_owned<F, NewType2>(self, f: F) -> Self::BinaryOutput<Self::Source, NewType2>
79//!     where
80//!         F: Fn(Self::Source2) -> NewType2,
81//!     {
82//!         BiTuple(self.0, f(self.1))
83//!     }
84//! }
85//!
86//! impl<A, B> Bifunctor for BiTuple<A, B>
87//! where
88//!     A: Clone,
89//!     B: Clone,
90//! {
91//!     fn first<C, F>(&self, f: F) -> BiTuple<C, B>
92//!     where
93//!         F: Fn(&A) -> C,
94//!     {
95//!         BiTuple(f(&self.0), self.1.clone())
96//!     }
97//!
98//!     fn second<D, G>(&self, g: G) -> BiTuple<A, D>
99//!     where
100//!         G: Fn(&B) -> D,
101//!     {
102//!         BiTuple(self.0.clone(), g(&self.1))
103//!     }
104//!
105//!     fn bimap<C, D, F, G>(&self, f: F, g: G) -> BiTuple<C, D>
106//!     where
107//!         F: Fn(&A) -> C,
108//!         G: Fn(&B) -> D,
109//!     {
110//!         BiTuple(f(&self.0), g(&self.1))
111//!     }
112//! }
113//!
114//! // Usage
115//! let tuple = BiTuple(10, "hello");
116//! let mapped = tuple.bimap(|x| x * 2, |s| s.len());
117//! assert_eq!(mapped.0, 20);
118//! assert_eq!(mapped.1, 5);
119//! ```
120//!
121//! ## Relationship to Other Traits
122//!
123//! - **Functor**: A bifunctor is a generalization of a functor to two type parameters
124//! - **Profunctor**: While a bifunctor is covariant in both arguments, a profunctor is
125//!   contravariant in its first argument and covariant in its second

129/// A bifunctor is a type constructor that takes two type arguments and is a functor in both arguments.
130/// This means it provides a way to map functions over both type parameters independently or simultaneously.
131///
132/// # Laws
133///
134/// A valid bifunctor instance must satisfy these laws:
135///
136/// 1. Identity:
137///    ```text
138///    bimap(id, id) == id
139///    ```
140///
141/// 2. Composition:
142///    ```text
143///    bimap(f . g, h . i) == bimap(f, h) . bimap(g, i)
144///    ```
145///
146/// # Examples
147///
148/// Here's an example implementation for Result:
149/// ```rust
150/// use std::fmt::Debug;
151/// use rustica::traits::hkt::{HKT, BinaryHKT};
152/// use rustica::traits::bifunctor::Bifunctor;
153///
154/// // A wrapper for Result to implement HKT and Bifunctor
155/// #[derive(Debug, Clone, PartialEq)]
156/// struct BiResult<T, E>(Result<T, E>);
157///
158/// // HKT implementation for the first type parameter
159/// impl<T, E> HKT for BiResult<T, E> {
160///     type Source = T;
161///     type Output<U> = BiResult<U, E>;
162/// }
163///
164/// impl<T, E> BinaryHKT for BiResult<T, E> {
165///     type Source2 = E;
166///     type BinaryOutput<U, V> = BiResult<U, V>;
167///
168///     fn map_second<F, NewType2>(&self, f: F) -> Self::BinaryOutput<T, NewType2>
169///     where
170///         F: Fn(&Self::Source2) -> NewType2,
171///         T: Clone,
172///     {
173///         BiResult(match &self.0 {
174///             Ok(a) => Ok(a.clone()),
175///             Err(e) => Err(f(e)),
176///         })
177///     }
178///
179///     fn map_second_owned<F, NewType2>(self, f: F) -> Self::BinaryOutput<T, NewType2>
180///     where
181///         F: Fn(Self::Source2) -> NewType2,
182///     {
183///         BiResult(match self.0 {
184///             Ok(a) => Ok(a),
185///             Err(e) => Err(f(e)),
186///         })
187///     }
188/// }
189///
190/// impl<T: Clone, E: Clone> Bifunctor for BiResult<T, E> {
191///     fn first<C, F>(&self, f: F) -> Self::BinaryOutput<C, E>
192///     where
193///         F: Fn(&T) -> C,
194///     {
195///         BiResult(match &self.0 {
196///             Ok(a) => Ok(f(a)),
197///             Err(e) => Err(e.clone()),
198///         })
199///     }
200///
201///     fn second<D, G>(&self, g: G) -> Self::BinaryOutput<T, D>
202///     where
203///         G: Fn(&E) -> D,
204///     {
205///         BiResult(match &self.0 {
206///             Ok(a) => Ok(a.clone()),
207///             Err(e) => Err(g(e)),
208///         })
209///     }
210///
211///     fn bimap<C, D, F, G>(&self, f: F, g: G) -> Self::BinaryOutput<C, D>
212///     where
213///         F: Fn(&T) -> C,
214///         G: Fn(&E) -> D,
215///     {
216///         BiResult(match &self.0 {
217///             Ok(a) => Ok(f(a)),
218///             Err(e) => Err(g(e)),
219///         })
220///     }
221/// }
222///
223/// // Example usage:
224/// let success: BiResult<i32, &str> = BiResult(Ok(5));
225/// let error: BiResult<i32, &str> = BiResult(Err("error"));
226///
227/// // Transform the success value
228/// let doubled = success.first(|x| x * 2);
229/// assert_eq!(doubled, BiResult(Ok(10)));
230///
231/// // Transform the error value
232/// let mapped_error = error.second(|e| e.to_string());
233/// assert_eq!(mapped_error, BiResult(Err("error".to_string())));
234///
235/// // Transform both simultaneously
236/// let both_mapped = success.bimap(|x| x * 2, |e| e.to_string());
237/// assert_eq!(both_mapped, BiResult(Ok(10)));
238///
239/// // Chain operations
240/// let result = success
241///     .first(|x| x + 3)      // 5 -> 8
242///     .first(|x| x * 2)      // 8 -> 16
243///     .second(|e| e.to_string());
244/// assert_eq!(result, BiResult(Ok(16)));
245/// ```
246///
247/// # Common Use Cases
248///
249/// Bifunctors are particularly useful in these scenarios:
250///
251/// 1. Error Handling:
252///    - Transform both success and error values in Result types
253///    - Map error types to a common error type while preserving success values
254///
255/// 2. Data Processing:
256///    - Process pairs of values independently
257///    - Transform both components of a tuple simultaneously
258///
259/// 3. Type Conversion:
260///    - Convert between different error types in error handling
261///    - Transform data structures that contain two type parameters
262pub trait Bifunctor: BinaryHKT {
263    /// Maps a function over the first type parameter.
264    ///
265    /// This is similar to `fmap` for regular functors, but it operates on the first
266    /// type parameter while leaving the second unchanged.
267    ///
268    /// # Type Parameters
269    ///
270    /// * `C`: The new type for the first parameter after transformation
271    /// * `F`: The function type to apply to the first parameter
272    ///
273    /// # Arguments
274    ///
275    /// * `f`: Function to apply to the first type parameter
276    ///
277    /// # Returns
278    ///
279    /// A new bifunctor with the first type parameter transformed
280    ///
281    /// # Examples
282    ///
283    /// ```rust
284    /// use rustica::traits::bifunctor::Bifunctor;
285    /// use rustica::traits::hkt::{HKT, BinaryHKT};
286    ///
287    /// // A simple pair type
288    /// #[derive(Debug, PartialEq, Clone)]
289    /// struct Pair<A, B>(A, B);
290    ///
291    /// impl<A, B> HKT for Pair<A, B> {
292    ///     type Source = A;
293    ///     type Output<U> = Pair<U, B>;
294    /// }
295    ///
296    /// impl<A, B> BinaryHKT for Pair<A, B> {
297    ///     type Source2 = B;
298    ///     type BinaryOutput<U, V> = Pair<U, V>;
299    ///
300    ///     fn map_second<F, NewType2>(&self, f: F) -> Self::BinaryOutput<Self::Source, NewType2>
301    ///     where
302    ///         F: Fn(&Self::Source2) -> NewType2,
303    ///         Self::Source: Clone,
304    ///     {
305    ///         Pair(self.0.clone(), f(&self.1))
306    ///     }
307    ///
308    ///     fn map_second_owned<F, NewType2>(self, f: F) -> Self::BinaryOutput<Self::Source, NewType2>
309    ///     where
310    ///         F: Fn(Self::Source2) -> NewType2,
311    ///     {
312    ///         Pair(self.0, f(self.1))
313    ///     }
314    /// }
315    ///
316    /// impl<A, B> Bifunctor for Pair<A, B>
317    /// where
318    ///     A: Clone,
319    ///     B: Clone,
320    /// {
321    ///     fn first<C, F>(&self, f: F) -> Pair<C, B>
322    ///     where
323    ///         F: Fn(&A) -> C,
324    ///     {
325    ///         Pair(f(&self.0), self.1.clone())
326    ///     }
327    ///
328    ///     fn second<D, G>(&self, g: G) -> Pair<A, D>
329    ///     where
330    ///         G: Fn(&B) -> D,
331    ///     {
332    ///         Pair(self.0.clone(), g(&self.1))
333    ///     }
334    ///
335    ///     fn bimap<C, D, F, G>(&self, f: F, g: G) -> Pair<C, D>
336    ///     where
337    ///         F: Fn(&A) -> C,
338    ///         G: Fn(&B) -> D,
339    ///     {
340    ///         Pair(f(&self.0), g(&self.1))
341    ///     }
342    /// }
343    ///
344    /// let pair: Pair<i32, String> = Pair(5, "hello".to_string());
345    /// let mapped = pair.first(|x| x * 2);
346    /// assert_eq!(mapped, Pair(10, "hello".to_string()));
347    /// ```
348    fn first<C, F>(&self, f: F) -> Self::BinaryOutput<C, Self::Source2>
349    where
350        F: Fn(&Self::Source) -> C,
351        C: Clone;
352
353    /// Maps a function over the second type parameter.
354    ///
355    /// This is the counterpart to `first` that operates on the second type parameter
356    /// while leaving the first unchanged.
357    ///
358    /// # Type Parameters
359    ///
360    /// * `D`: The new type for the second parameter after transformation
361    /// * `G`: The function type to apply to the second parameter
362    ///
363    /// # Arguments
364    ///
365    /// * `f`: Function to apply to the second type parameter
366    ///
367    /// # Returns
368    ///
369    /// A new bifunctor with the second type parameter transformed
370    ///
371    /// # Examples
372    ///
373    /// ```rust
374    /// use rustica::traits::bifunctor::Bifunctor;
375    /// use rustica::traits::hkt::{HKT, BinaryHKT};
376    ///
377    /// // Using Either as an example bifunctor
378    /// #[derive(Debug, PartialEq, Clone)]
379    /// enum Either<L, R> {
380    ///     Left(L),
381    ///     Right(R),
382    /// }
383    ///
384    /// impl<L, R> HKT for Either<L, R> {
385    ///     type Source = L;
386    ///     type Output<U> = Either<U, R>;
387    /// }
388    ///
389    /// impl<L, R> BinaryHKT for Either<L, R> {
390    ///     type Source2 = R;
391    ///     type BinaryOutput<U, V> = Either<U, V>;
392    ///
393    ///     fn map_second<F, NewType2>(&self, f: F) -> Self::BinaryOutput<Self::Source, NewType2>
394    ///     where
395    ///         F: Fn(&Self::Source2) -> NewType2,
396    ///         Self::Source: Clone,
397    ///     {
398    ///         match self {
399    ///             Either::Left(l) => Either::Left(l.clone()),
400    ///             Either::Right(r) => Either::Right(f(r)),
401    ///         }
402    ///     }
403    ///
404    ///     fn map_second_owned<F, NewType2>(self, f: F) -> Self::BinaryOutput<Self::Source, NewType2>
405    ///     where
406    ///         F: Fn(Self::Source2) -> NewType2,
407    ///     {
408    ///         match self {
409    ///             Either::Left(l) => Either::Left(l),
410    ///             Either::Right(r) => Either::Right(f(r)),
411    ///         }
412    ///     }
413    /// }
414    ///
415    /// impl<L, R> Bifunctor for Either<L, R>
416    /// where
417    ///     L: Clone,
418    ///     R: Clone,
419    /// {
420    ///     fn first<C, F>(&self, f: F) -> Either<C, R>
421    ///     where
422    ///         F: Fn(&L) -> C,
423    ///     {
424    ///         match self {
425    ///             Either::Left(l) => Either::Left(f(l)),
426    ///             Either::Right(r) => Either::Right(r.clone()),
427    ///         }
428    ///     }
429    ///
430    ///     fn second<D, G>(&self, g: G) -> Either<L, D>
431    ///     where
432    ///         G: Fn(&R) -> D,
433    ///     {
434    ///         match self {
435    ///             Either::Left(l) => Either::Left(l.clone()),
436    ///             Either::Right(r) => Either::Right(g(r)),
437    ///         }
438    ///     }
439    ///
440    ///     fn bimap<C, D, F, G>(&self, f: F, g: G) -> Either<C, D>
441    ///     where
442    ///         F: Fn(&L) -> C,
443    ///         G: Fn(&R) -> D,
444    ///     {
445    ///         match self {
446    ///             Either::Left(l) => Either::Left(f(l)),
447    ///             Either::Right(r) => Either::Right(g(r)),
448    ///         }
449    ///     }
450    /// }
451    ///
452    /// let right: Either<i32, String> = Either::Right("hello".to_string());
453    /// let mapped = right.second(|s| s.len());
454    /// assert_eq!(mapped, Either::Right(5));
455    /// ```
456    fn second<D, G>(&self, f: G) -> Self::BinaryOutput<Self::Source, D>
457    where
458        G: Fn(&Self::Source2) -> D,
459        D: Clone;
460
461    /// Maps two functions over both type parameters simultaneously.
462    ///
463    /// This combines the functionality of `first` and `second` into a single operation.
464    /// It's equivalent to applying `first` followed by `second`, but may be more efficient.
465    ///
466    /// # Type Parameters
467    ///
468    /// * `C`: The new type for the first parameter after transformation
469    /// * `D`: The new type for the second parameter after transformation
470    /// * `F`: The function type to apply to the first parameter
471    /// * `G`: The function type to apply to the second parameter
472    ///
473    /// # Arguments
474    ///
475    /// * `f`: Function to apply to the first type parameter
476    /// * `g`: Function to apply to the second type parameter
477    ///
478    /// # Returns
479    ///
480    /// A new bifunctor with both type parameters transformed
481    ///
482    /// # Examples
483    ///
484    /// ```rust
485    /// use rustica::traits::bifunctor::Bifunctor;
486    /// use rustica::traits::hkt::{HKT, BinaryHKT};
487    ///
488    /// #[derive(Debug, PartialEq, Clone)]
489    /// struct Pair<A, B>(A, B);
490    ///
491    /// impl<A, B> HKT for Pair<A, B> {
492    ///     type Source = A;
493    ///     type Output<U> = Pair<U, B>;
494    /// }
495    ///
496    /// impl<A, B> BinaryHKT for Pair<A, B> {
497    ///     type Source2 = B;
498    ///     type BinaryOutput<U, V> = Pair<U, V>;
499    ///
500    ///     fn map_second<F, NewType2>(&self, f: F) -> Self::BinaryOutput<Self::Source, NewType2>
501    ///     where
502    ///         F: Fn(&Self::Source2) -> NewType2,
503    ///         Self::Source: Clone,
504    ///     {
505    ///         Pair(self.0.clone(), f(&self.1))
506    ///     }
507    ///
508    ///     fn map_second_owned<F, NewType2>(self, f: F) -> Self::BinaryOutput<Self::Source, NewType2>
509    ///     where
510    ///         F: Fn(Self::Source2) -> NewType2,
511    ///     {
512    ///         Pair(self.0, f(self.1))
513    ///     }
514    /// }
515    ///
516    /// impl<A, B> Bifunctor for Pair<A, B>
517    /// where
518    ///     A: Clone,
519    ///     B: Clone,
520    /// {
521    ///     fn first<C, F>(&self, f: F) -> Pair<C, B>
522    ///     where
523    ///         F: Fn(&A) -> C,
524    ///     {
525    ///         Pair(f(&self.0), self.1.clone())
526    ///     }
527    ///
528    ///     fn second<D, G>(&self, g: G) -> Pair<A, D>
529    ///     where
530    ///         G: Fn(&B) -> D,
531    ///     {
532    ///         Pair(self.0.clone(), g(&self.1))
533    ///     }
534    ///
535    ///     fn bimap<C, D, F, G>(&self, f: F, g: G) -> Pair<C, D>
536    ///     where
537    ///         F: Fn(&A) -> C,
538    ///         G: Fn(&B) -> D,
539    ///     {
540    ///         Pair(f(&self.0), g(&self.1))
541    ///     }
542    /// }
543    ///
544    /// let pair: Pair<i32, String> = Pair(5, "hello".to_string());
545    /// let mapped = pair.bimap(|x| x * 2, |s| s.len());
546    /// assert_eq!(mapped, Pair(10, 5));
547    ///
548    /// // Equivalent to chaining first and second:
549    /// let alternative = pair
550    ///     .first(|x| x * 2)
551    ///     .second(|s| s.len());
552    /// assert_eq!(alternative, Pair(10, 5));
553    /// ```
554    fn bimap<C, D, F, G>(&self, f: F, g: G) -> Self::BinaryOutput<C, D>
555    where
556        F: Fn(&Self::Source) -> C,
557        G: Fn(&Self::Source2) -> D,
558        C: Clone,
559        D: Clone;
560}
